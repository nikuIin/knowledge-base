[https://habr.com/ru/articles/192228/] — пункт 2) создание таблиц локализации


Если нам нужно поддерживать разные языки для объекта, то создаем 2 таблицы:
- таблица с независимыми от языка данными
- таблица с зависимыми от языка данными
- таблицу языков (если она еще не создана)

```sql
create table country (
  country_id int primary key generated always as identity,
  flag_id int references flag(flag_id)
);

create table country_translate(
  country_id integer not null references country(country_id),
  language_id varchar(10) not null references language(language_id),
  name varchar(255) not null,
  primary key(country_id, language_id)
);

create table language(
  language_id varchar(10), -- short language code e.g. en-US
  name varchar(255), -- full language name e.g. American Enlish,
  flag_id int references flag(flag_id)
);
```
