
1. Create `TABLESPACES` for this types of work [[tablespace]]:
	* for fast I/O-operations (SSD)
	* for cheap I/O-operations (HDD)

```shell
mkdir /ssd1/postgresql/data/...
mkdir /hhd1/postgresql/data/...
```

```sql
-- create tablespaces
create tablespace fastspace location '/ssd1/postgresql/data';
create tablespace default_space location '/hhd1/postgresql/data';

-- set default table space
set default_tablespace = default_space;

-- or you can explicitly state it
create table some_table (...) tablespace default_space;
```

2. Create table and add partition on it
```sql
create table some_table(
	some_table int...,
	date date not null default now()::date,
	some_data jsonb
	primary_key(some_table, date) -- column of partition must be included in the primary key
) partition by range (date);

create table some_table_2020 partition of some_table
for values from ('2020-01-01') to ('2021-01-01')
tablespace default_space;

-- create all other partitions
-- ...

-- create default (now) partition on fast tablespace
create table some_table_current_year partition of some_table default
tablespace fastspace;

```
